Create src/components/layout/NewsletterForm.tsx as a 'use client' component. It captures first name + email, POSTs to /api/newsletter/subscribe, and shows inline feedback based on response status.

Shape:

```typescript
'use client'

import { useState } from 'react'
import { usePathname } from 'next/navigation'

type Status = 'idle' | 'loading' | 'success' | 'duplicate' | 'error'

export function NewsletterForm() {
  const pathname = usePathname()
  const [firstName, setFirstName] = useState('')
  const [email, setEmail] = useState('')
  const [status, setStatus] = useState<Status>('idle')

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    setStatus('loading')
    try {
      const res = await fetch('/api/newsletter/subscribe', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ first_name: firstName, email, source_page: pathname }),
      })
      if (res.status === 409) setStatus('duplicate')
      else if (res.ok) setStatus('success')
      else setStatus('error')
    } catch {
      setStatus('error')
    }
  }
  // render...
}
```

Render structure:
- A `<form onSubmit={handleSubmit}>` containing:
  - First name input: `type="text"` required, placeholder "Your first name", value/onChange bound to firstName state
  - Email input: `type="email"` required, placeholder "Your email address", value/onChange bound to email state
  - Submit button: label "Join", disabled when status === 'loading', shows "Joining..." text when loading
- Below the form, conditionally render one message based on status:
  - success: "You're in! Welcome to the Twinkle family."
  - duplicate: "Looks like you're already on the list!"
  - error: "Something went wrong. Please try again."

Styling must work on the dark bg-cocoa footer background:
- Inputs: `w-full rounded-md bg-cocoa border border-cream/20 text-cream placeholder:text-cream/40 px-3 py-2 text-sm focus:outline-none focus:border-gold font-body`
- Submit button: `w-full mt-2 rounded-md bg-gold text-cocoa font-heading font-semibold text-sm py-2 px-4 hover:bg-gold/90 transition-colors disabled:opacity-60 disabled:cursor-not-allowed`
- Success message: `text-sm font-body text-gold mt-3`
- Duplicate message: `text-sm font-body text-cream/70 mt-3`
- Error message: `text-sm font-body text-terracotta mt-3`
- Stack first name above email (flex-col gap-2)

Use usePathname() from next/navigation for source_page — do NOT use window.location.href (SSR unsafe).
