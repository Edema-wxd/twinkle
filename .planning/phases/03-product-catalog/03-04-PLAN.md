    If type generation fails (CLI not linked), manually update src/types/supabase.ts to add the products table row type to the Database interface. The minimum needed is that the Database type includes public.Tables.products.Row with the correct column types.

    Step 2: Swap the mock import for Supabase in src/app/catalog/page.tsx:

    Replace the existing synchronous Server Component with an async one:

    ```typescript
    import { createClient } from '@/lib/supabase/server'
    import { CatalogClient } from '@/components/catalog/CatalogClient'
    import { Product, ProductVariant } from '@/lib/types/product'

    export const metadata = {
      title: 'Shop Loc Beads — Twinkle Locs',
      description: 'Browse all premium loc bead accessories. Filter by material, search by name.',
    }

    export default async function CatalogPage() {
      const supabase = await createClient()
      const { data, error } = await supabase
        .from('products')
        .select('*')
        .order('created_at', { ascending: false })

      if (error) {
        console.error('Failed to fetch products:', error.message)
      }

      // Cast variants JSONB to typed ProductVariant[] (avoids Json type mismatch)
      const products: Product[] = (data ?? []).map(row => ({
        ...row,
        variants: row.variants as unknown as ProductVariant[],
      }))

      return <CatalogClient products={products} />
    }
    ```
