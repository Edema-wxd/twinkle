  <files>src/app/catalog/[slug]/page.tsx</files>
  <action>
    Update src/app/catalog/[slug]/page.tsx to add two more Supabase queries and pass data to the new components.

    Add after the existing product fetch:

    ```typescript
    // Fetch reviews for this product
    const { data: reviewsData } = await supabase
      .from('reviews')
      .select('*')
      .eq('product_id', product.id)
      .order('created_at', { ascending: false })

    const reviews = (reviewsData ?? []) as Review[]

    // Fetch shears for upsell — only when viewing a non-Tools product
    let shearsProduct: Product | null = null
    if (product.material !== 'Tools') {
      const { data: shearsData } = await supabase
        .from('products')
        .select('*')
        .eq('slug', 'shears')
        .single()

      if (shearsData) {
        shearsProduct = {
          ...shearsData,
          variants: shearsData.variants as unknown as ProductVariant[],
          images: shearsData.images?.length ? shearsData.images : undefined,
        }
      }
    }
    ```

    Add imports: `Review` from `@/lib/types/review`, `ProductReviews` from `@/components/product/ProductReviews`, `UpsellBlock` from `@/components/product/UpsellBlock`.

    Update the return JSX to add reviews and upsell below the product grid:

      {/* Product detail grid (ProductDetailClient) */}

      {/* Reviews section */}
